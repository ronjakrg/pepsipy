49def extract_related_kwargs(mapping: dict, params: dict) -> dict: 

50 """ 

51 Extracts entries whose external key appears in a given mapping from a dictionary. The found entries are returned with their internal key and values with None are ignored. 

52 mapping: Dictionary which maps external keys to internal keys 

53 params: Dictionary containing external keys and values 

54 """

63def convert_exponential_to_suffix(exp: int) -> str: 

64 """ 

65 Converts an exponent (e.g., 6 for 10e6) to a human-readable suffix (e.g., 1M) 

66 """ 

67 value = 10**exp 

68 if value >= 10**12: 

69 return f"{value//10**12}T" 

70 elif value >= 10**9: 

71 return f"{value//10**9}B" 

72 elif value >= 10**6: 

73 return f"{value//10**6}M" 

74 elif value >= 10**3:

76 else:
